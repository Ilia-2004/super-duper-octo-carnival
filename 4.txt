using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Введите колличество щелчков: ");

            int n = int.Parse(Console.ReadLine());
            int i = 0;
            int hours = 12, min = 30;

            float angle = 0, angle1 = 165, hourAngle = 0, minAngle = 0;

            while (i != n)
            {
                if (min == 60) 
                { 
                    hours++; min = 0; 
                }

                min++;

                hourAngle = 0.5f * (hours * 60 + min);
                minAngle = 6 * min;

                angle = Math.Abs(hourAngle - minAngle);
                angle = Math.Min(angle, 360 - angle);

                if (angle1 < 30 && angle >= 30) 
                { 
                    i++; 
                }

                angle1 = angle;
            }

            Console.WriteLine();

            Console.WriteLine("Ответ: {0}:{1}", hours, min);

            Console.WriteLine();
        }
    }
}
